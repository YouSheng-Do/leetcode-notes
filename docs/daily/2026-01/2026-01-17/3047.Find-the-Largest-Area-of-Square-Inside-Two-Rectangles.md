```cpp
class Solution {
public:
    long long largestSquareArea(vector<vector<int>>& bottomLeft, vector<vector<int>>& topRight) {
        int maxSide = 0;
        const int n = bottomLeft.size();

        for (int i = 0; i < n - 1; ++i) {
            for (int j = i + 1; j < n; ++j) {
                int xSide = min(topRight[i][0], topRight[j][0]) - max(bottomLeft[i][0], bottomLeft[j][0]);
                int ySide = min(topRight[i][1], topRight[j][1]) - max(bottomLeft[i][1], bottomLeft[j][1]);
                if (xSide > 0 && ySide > 0) {
                    maxSide = max(maxSide, min(xSide, ySide));
                }
            }
        }

        return 1LL * maxSide * maxSide;
    }
};
```
